    def annotate_scaled(img, width, height, x, y, text, scale)
      scaled_width = (width * scale) >= 1 ? (width * scale) : 1
      scaled_height = (height * scale) >= 1 ? (height * scale) : 1

      self.annotate( img,
                     scaled_width, scaled_height,
                     x * scale, y * scale,
                     text)
    end
