      def search *paths
        ns = paths.last.is_a?(Hash) ? paths.pop :
          (document.root ? document.root.namespaces : {})

        sub_set = NodeSet.new(document)
        each do |node|
          paths.each do |path|
            sub_set +=
              send(path =~ /^(\.\/|\/)/ ? :xpath : :css, *(paths + [ns]))
          end
        end
        document.decorate(sub_set)
        sub_set
      end
