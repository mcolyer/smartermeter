      def initialize document, tags=nil
        if tags
          parser = if self.kind_of?(Nokogiri::HTML::DocumentFragment)
                     HTML::SAX::Parser.new(FragmentHandler.new(self, tags))
                   else
                     XML::SAX::Parser.new(FragmentHandler.new(self, tags))
                   end
          parser.parse(tags)
        end
      end
